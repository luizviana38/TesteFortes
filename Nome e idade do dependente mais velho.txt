
SELECT A.NOME EMPREGADO,
       C.DEPEN_NOME DEP_MAIS_VELHO,
	   convert(varchar(10),C.DATA_NASCIMENTO,103) NASCIMENTO
  FROM EMPREGADO A, 
       (SELECT T.RG_RESP,  MIN(T.DATA_NASCIMENTO) NASCIMENTO 
        FROM DEPENDENTE T
        GROUP BY T.RG_RESP ) B,
		DEPENDENTE C
WHERE A.RG = B.RG_RESP
 AND  A.RG = C.RG_RESP
 AND B.NASCIMENTO = C.DATA_NASCIMENTO
 ORDER BY A.NOME
